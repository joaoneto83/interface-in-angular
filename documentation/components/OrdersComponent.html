import { Component, OnInit } from '@angular/core';
import { dateMask } from 'src/shared/functions/form-functions';

@Component({
  selector: 'elgin-orders',
  templateUrl: './orders.component.html',
  styleUrls: ['./orders.component.scss']
})
export class OrdersComponent implements OnInit {

  active = 1;

  from: string;
  to: string;
  customFilter: string;

  filterObjects: any = [
    'Representante-Geral',
    'Varejo'
  ];

  storesObjects: any = [
    'Magazine Luiza',
    'Casas Bahia',
    'Ponto Frio',
    'Elgin Industrial Amazonia LTDA',
    'Elgin SA 02',
    'suprimentos',
    'em carteira'
  ];

  constructor() { }

  ngOnInit(): void {
    this.customFilter = this.filterObjects[0];
  }

  selectType(e?: any) {
    this.customFilter = e.target.value;
    console.log(this.customFilter);
  }

  regex(e: any) {
    const input: HTMLInputElement = e.target;

    input.value = dateMask(input.value);
  }

}
