import { Component, OnInit, Output, EventEmitter } from '@angular/core';

@Component({
  selector: 'elgin-side-menu',
  templateUrl: './side-menu.component.html',
  styleUrls: ['./side-menu.component.scss']
})
export class SideMenuComponent implements OnInit {

  @Output() sideMenuData = new EventEmitter();

  routeId: number;
  routeTitle: string;

  selectedItem = 1;

  menuItems = [
    // { icon: 'fa fa-user', title: null, id: 0 },
    { icon: 'fa fa-home', title: 'Inicial', id: 1 },
    { icon: 'fa fa-cart-arrow-down', title: 'Pedidos', id: 2 },
    { icon: 'fa fa-box-open', title: 'Produtos', id: 3 },
    { icon: 'fa fa-laptop', title: 'Gerencial', id: 4 },
    { icon: 'fa fa-suitcase', title: 'Representantes', id: 5 },
    { icon: 'fa fa-users', title: 'Clientes', id: 6 },
    { icon: 'fa fa-file-invoice-dollar', title: 'Financeiro', id: 7 },
    { icon: 'fa fa-user-tie', title: 'Poítica Comercial', id: 8 },
    { icon: 'fa fa-file-alt', title: 'Relatórios', id: 9 },
    { icon: 'fa fa-cog', title: 'Admin', id: 10 },
  ];

  constructor() { }

  ngOnInit() { }

  emitEventData() {
    const data = { id: this.routeId, title: this.routeTitle };
    this.sideMenuData.emit(data);
  }

  hideSideMenu() {
    const hover = document.getElementById('hoverMenu');

    if (hover.classList.contains('hover')) {
      hover.classList.remove('hover');

      setTimeout(() => {
        hover.classList.add('hover');
      }, 800);
    }
  }

  navigate(index: number) {
    this.selectedItem = index;
    this.routeId = this.menuItems[this.selectedItem].id;
    this.routeTitle = this.menuItems[this.selectedItem].title;

    console.log(
      'INDEX ==>', index,
      'item selecionado ==>', this.selectedItem,
      'Id da rota ==>', this.routeId,
      'Titulo da rota ==>', this.routeTitle
    );

    this.emitEventData();

    this.hideSideMenu();
  }

}
